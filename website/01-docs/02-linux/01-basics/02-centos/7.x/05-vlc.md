# Инсталляция VLC в Centos 7

<br/>

    # rpm -Uvh http://li.nux.ro/download/nux/dextop/el7/x86_64/nux-dextop-release-0-5.el7.nux.noarch.rpm

<br/>

    # yum install -y vlc


<br/>

-- обновить

    # yum clean all && yum update
